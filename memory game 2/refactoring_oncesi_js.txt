/* 1. adım  */ 
const cards = document.querySelectorAll('.memory-card');

/*4. adım */
let hasFlippedCard = false;
let firstCard, secondCard;


/* 3. adım */
function flipCard(){
    /* console.log("clicked");
    console.log(this);  */
    /* 4. adımda aşağıdaki toggle ı değiştirdim  */
    /* this.classList.toggle('flip'); */
    this.classList.add('flip');
    /* 4. adım devam */
    if(!hasFlippedCard) {
        //first click
        hasFlippedCard = true;
        firstCard = this;
        /* console.log({hasFlippedCard, firstCard}); */
    } else {
        //second click
        hasFlippedCard = false;
        secondCard = this;
        /* console.log({firstCard, secondCard}); */
        // do cards match?
        /*
        iki card ı karşılaştırmak için html de çok kulanışlı bir attribute var data attr
        html ye bunu eklediğimizde buradan js kodu içinden karşılaştırma yapabiliriz
        data-* şeklinde kullanımı var
        */
       /* console.log(firstCard.dataset.framework);
       console.log(secondCard.dataset.framework); */
       if(firstCard.dataset.framework === secondCard.dataset.framework) {
           firstCard.removeEventListener('click', flipCard);
           secondCard.removeEventListener('click', flipCard);
       }
       /* console.log("eşleşme oldu"); */
       else {
           /* not matching */
           setTimeout(() => {
            firstCard.classList.remove('flip');
            secondCard.classList.remove('flip');
        }, 1500);           
       }
   }
}

/* 2. adım */
cards.forEach(card => card.addEventListener('click', flipCard));
